View: dbo.Vedanta_User_Cat_Plant_Checkpoint
Definition:

CREATE VIEW [dbo].[Vedanta_User_Cat_Plant_Checkpoint]
AS
SELECT distinct pramod.*, ISNULL(cp.Location, '') AS Location, ISNULL(cp.Latitude, '') AS Latitude, ISNULL(cp.Longitude, '') AS Longitude, ISNULL(cp.GeofenceType, '') AS GeofenceType, ISNULL(cp.Radius, '')

FROM            (SELECT        LTRIM(RTRIM(plant)) PlantName, PlantID PlantID, CatName[CategoryName], [Ecode] UserID, CatCode CategoryID, t2.Data, [Type] PetrollType, TypeID[PetrollTypeID], TaskStartDate, EndDate
                          FROM            [vw_ObservationRound] CROSS apply
                                                        (SELECT        Data
                                                          FROM            dbo.Split([vw_ObservationRound].CheckPointIDs, ',')) t2 ) pramod INNER JOIN
                         RoundCheckPoint_RTL rtl ON pramod.Data = rtl.CheckPointID INNER JOIN
                         Vedanta_Plant_Checkpoint cp ON rtl.CheckPointID = cp.RecordID

View: dbo.Vedanta_User_Cat_Plant_Checkpoint2
Definition:
CREATE VIEW [dbo].[Vedanta_User_Cat_Plant_Checkpoint2]
AS
select '' [CategoryName], obs.Ecode UserID, 
obs.RecordID CategoryID, 0 [Data], otype.TypeName PetrollType, TypeID [PetrollTypeID], TaskStartDate, EndDate,
ISNULL(cp.Location, '') AS Location, ISNULL(cp.Latitude, '') AS Latitude, 
ISNULL(cp.Longitude, '') AS Longitude, ISNULL(cp.GeofenceType, '') AS GeofenceType, ISNULL(cp.Radius, '') AS Radius, 
ISNULL(cp.Lat_Lng_Lst, '') AS Lat_Lng_Lst, cp.RecordID AS CheckPointID, 
cp.CheckPointName, cp.Severity, ISNULL(cp.Call_Sign, '') AS Call_Sign,
LTRIM(RTRIM(PlantName)) PlantName, plant.RecordID PlantID
from Vedanta_Observation_Round obs 
inner join Vedanta_Observation_Type otype on otype.RecordID = obs.TypeID
inner join GTMSTaskGroup GTP on obs.TaskGroupID=GTP.TaskGroupID
inner join RoundCheckPoint_RTL rcp on rcp.RoundID=obs.RecordID
INNER JOIN Vedanta_Plant_Checkpoint cp ON rcp.CheckPointID = cp.RecordID
INNER JOIN Vedanta_Plant_Location plant ON plant.RecordID = cp.plantid
